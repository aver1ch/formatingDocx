        self.logger.info("–ü—Ä–∏–º–µ–Ω—è—é –º–Ω–æ–≥–æ—É—Ä–æ–≤–Ω–µ–≤—É—é –Ω—É–º–µ—Ä–∞—Ü–∏—é —Ä–∞–∑–¥–µ–ª–æ–≤...")

            style = paragraph.style.name
            
            if style not in self.heading_levels:

            level = self.heading_levels[style]

            self._add_numbering_to_paragraph(paragraph, level)
    
    def _update_section_number(self, level: int) -> None:
        """–û–±–Ω–æ–≤–∏—Ç—å —Å—á–µ—Ç—á–∏–∫ –¥–ª—è —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ —É—Ä–æ–≤–Ω—è."""

        """–î–æ–±–∞–≤–∏—Ç—å –Ω–æ–º–µ—Ä –∫ –∞–±–∑–∞—Ü—É (–∑–∞–≥–æ–ª–æ–≤–∫—É)."""

        paragraph.clear()
        
        # –î–æ–±–∞–≤–∏—Ç—å –Ω–æ–º–µ—Ä

        run.bold = True
        
        # –î–æ–±–∞–≤–∏—Ç—å –æ—Å—Ç–∞–ª—å–Ω–æ–π —Ç–µ–∫—Å—Ç
        run = paragraph.add_run(current_text)
        run.bold = True

        """–ü–æ–ª—É—á–∏—Ç—å –æ—Ç—Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –Ω–æ–º–µ—Ä —Ä–∞–∑–¥–µ–ª–∞."""

        return ".".join(str(n) for n in numbers)
